DatabaseUtils.java
import java.sql.*;

public class DatabaseUtils {
    private static final String URL = "jdbc:mysql://localhost:3306/QuanLyBanHang?useSSL=false&serverTimezone=UTC";
    private static final String USERNAME = "root";
    private static final String PASSWORD = "123456";
    
    public static Connection getConnection() throws SQLException {
        return DriverManager.getConnection(URL, USERNAME, PASSWORD);
    }
    
    public static void close(Connection conn) {
        try { if (conn != null) conn.close(); } catch (SQLException e) {}
    }
    
    public static void close(Statement stmt) {
        try { if (stmt != null) stmt.close(); } catch (SQLException e) {}
    }
    
    public static void close(ResultSet rs) {
        try { if (rs != null) rs.close(); } catch (SQLException e) {}
    }
    
    public static void close(Connection conn, Statement stmt, ResultSet rs) {
        close(rs);
        close(stmt);
        close(conn);
    }
}

PHẦN 1: TRANSACTION MANAGEMENT

1.1 TRANSACTION LÀ GÌ?

Định nghĩa:

Transaction (giao dịch) là một nhóm các câu lệnh SQL được thực thi như một đơn vị công việc duy nhất. Hoặc tất cả thành công, hoặc tất cả thất bại.

Ví dụ thực tế:

Chuyển tiền từ tài khoản A sang tài khoản B:

  1. Trừ tiền tài khoản A
  2. Cộng tiền tài khoản B

Nếu bước 1 thành công, bước 2 thất bại → tiền sẽ mất! Cần transaction để đảm bảo cả hai cùng thành công hoặc cùng thất bại.

1.2 TÍNH CHẤT ACID CỦA TRANSACTION

Tính chất Tên đầy đủ Mô tả
A Atomicity Toàn bộ transaction thành công hoặc thất bại hoàn toàn
C Consistency Dữ liệu luôn hợp lệ trước và sau transaction
I Isolation Các transaction độc lập, không ảnh hưởng lẫn nhau
D Durability Dữ liệu đã commit được lưu vĩnh viễn

1.3 CÁC PHƯƠNG THỨC QUẢN LÝ TRANSACTION TRONG JDBC

Phương thức Mô tả
setAutoCommit(boolean autoCommit) Bật/tắt chế độ tự động commit
commit() Xác nhận transaction, lưu vĩnh viễn
rollback() Hủy bỏ transaction, quay lại trạng thái ban đầu
setSavepoint(String name) Đánh dấu điểm khôi phục
rollback(Savepoint sp) Khôi phục về điểm đã đánh dấu

1.4 VÍ DỤ CƠ BẢN VỀ TRANSACTION

Ví dụ: Mua hàng - cập nhật số lượng tồn kho và thêm hóa đơn

TransactionExample.java
import java.sql.*;
import java.time.LocalDate;

public class TransactionExample {
    public static void main(String[] args) {
        Connection conn = null;
        
        try {
            // 1. Kết nối database
            conn = DatabaseUtils.getConnection();
            
            // 2. Tắt auto-commit (bắt đầu transaction)
            conn.setAutoCommit(false);
            
            // 3. Thực hiện các thao tác
            // 3.1. Thêm hóa đơn mới
            String insertHD = "INSERT INTO HoaDon (NgayLap, TongTien) VALUES (?, ?)";
            PreparedStatement pstmt1 = conn.prepareStatement(insertHD, Statement.RETURN_GENERATED_KEYS);
            pstmt1.setDate(1, Date.valueOf(LocalDate.now()));
            pstmt1.setDouble(2, 17000000); // 15tr + 2tr
            pstmt1.executeUpdate();
            
            // Lấy mã hóa đơn vừa tạo
            ResultSet rs = pstmt1.getGeneratedKeys();
            int maHD = 0;
            if (rs.next()) {
                maHD = rs.getInt(1);
            }
            
            // 3.2. Thêm chi tiết hóa đơn
            String insertCT = "INSERT INTO ChiTietHD (MaHD, MaSP, SoLuong, DonGia) VALUES (?, ?, ?, ?)";
            PreparedStatement pstmt2 = conn.prepareStatement(insertCT);
            
            // Sản phẩm 1: Laptop (số lượng 1)
            pstmt2.setInt(1, maHD);
            pstmt2.setInt(2, 1);
            pstmt2.setInt(3, 1);
            pstmt2.setDouble(4, 15000000);
            pstmt2.executeUpdate();
            
            // Sản phẩm 2: Tai nghe (số lượng 1)
            pstmt2.setInt(1, maHD);
            pstmt2.setInt(2, 3);
            pstmt2.setInt(3, 1);
            pstmt2.setDouble(4, 2000000);
            pstmt2.executeUpdate();
            
            // 3.3. Cập nhật số lượng tồn kho
            String updateSP = "UPDATE SanPham SET SoLuongTon = SoLuongTon - ? WHERE MaSP = ?";
            PreparedStatement pstmt3 = conn.prepareStatement(updateSP);
            
            // Giảm số lượng Laptop
            pstmt3.setInt(1, 1);
            pstmt3.setInt(2, 1);
            pstmt3.executeUpdate();
            
            // Giảm số lượng Tai nghe
            pstmt3.setInt(1, 1);
            pstmt3.setInt(2, 3);
            pstmt3.executeUpdate();
            
            // 4. Nếu mọi thứ OK, commit transaction
            conn.commit();
            System.out.println("Mua hàng thành công! Mã hóa đơn: " + maHD);
            
        } catch (SQLException e) {
            // 5. Nếu có lỗi, rollback transaction
            try {
                if (conn != null) {
                    conn.rollback();
                    System.out.println("Đã rollback - giao dịch thất bại!");
                }
            } catch (SQLException ex) {
                ex.printStackTrace();
            }
            e.printStackTrace();
            
        } finally {
            DatabaseUtils.close(conn);
        }
    }
}

1.5 VÍ DỤ VỚI SAVEPOINT

Mục đích:

Savepoint cho phép đánh dấu các điểm khôi phục trong transaction, có thể rollback về điểm đó mà không hủy toàn bộ.

SavepointExample.java
import java.sql.*;

public class SavepointExample {
    public static void main(String[] args) {
        Connection conn = null;
        Savepoint sp1 = null;
        Savepoint sp2 = null;
        
        try {
            conn = DatabaseUtils.getConnection();
            conn.setAutoCommit(false);
            
            // Thao tác 1: Thêm sản phẩm mới
            String insertSP = "INSERT INTO SanPham (TenSP, DonGia, SoLuongTon) VALUES (?, ?, ?)";
            PreparedStatement pstmt = conn.prepareStatement(insertSP);
            
            pstmt.setString(1, "Màn hình Samsung");
            pstmt.setDouble(2, 5000000);
            pstmt.setInt(3, 20);
            pstmt.executeUpdate();
            
            // Đánh dấu savepoint 1
            sp1 = conn.setSavepoint("AfterInsertProduct");
            System.out.println("Đã thêm sản phẩm");
            
            // Thao tác 2: Thêm hóa đơn
            String insertHD = "INSERT INTO HoaDon (NgayLap, TongTien) VALUES (?, ?)";
            pstmt = conn.prepareStatement(insertHD, Statement.RETURN_GENERATED_KEYS);
            pstmt.setDate(1, Date.valueOf("2024-03-15"));
            pstmt.setDouble(2, 5000000);
            pstmt.executeUpdate();
            
            ResultSet rs = pstmt.getGeneratedKeys();
            int maHD = 0;
            if (rs.next()) {
                maHD = rs.getInt(1);
            }
            
            // Đánh dấu savepoint 2
            sp2 = conn.setSavepoint("AfterInsertInvoice");
            System.out.println("Đã thêm hóa đơn");
            
            // Thao tác 3: Thêm chi tiết hóa đơn (giả sử có lỗi)
            String insertCT = "INSERT INTO ChiTietHD (MaHD, MaSP, SoLuong, DonGia) VALUES (?, ?, ?, ?)";
            pstmt = conn.prepareStatement(insertCT);
            
            pstmt.setInt(1, maHD);
            pstmt.setInt(2, 999); // Mã sản phẩm không tồn tại
            pstmt.setInt(3, 1);
            pstmt.setDouble(4, 5000000);
            pstmt.executeUpdate(); // Sẽ gây lỗi foreign key
            
            // Nếu thành công
            conn.commit();
            
        } catch (SQLException e) {
            try {
                if (conn != null) {
                    // Rollback về savepoint sau khi thêm hóa đơn
                    if (sp2 != null) {
                        conn.rollback(sp2);
                        System.out.println("Đã rollback về sau khi thêm hóa đơn");
                        
                        // Commit phần đã thêm sản phẩm và hóa đơn
                        conn.commit();
                        System.out.println("Đã commit phần sản phẩm và hóa đơn");
                        
                    } else if (sp1 != null) {
                        conn.rollback(sp1);
                        conn.commit();
                        System.out.println("Đã rollback về sau khi thêm sản phẩm");
                    
                    } else {
                        conn.rollback();
                        System.out.println("Đã rollback toàn bộ");
                    }
                }
            } catch (SQLException ex) {
                ex.printStackTrace();
            }
            e.printStackTrace();
            
        } finally {
            DatabaseUtils.close(conn);
        }
    }
}

1.6 ISOLATION LEVELS (MỨC ĐỘ CÔ LẬP)

Định nghĩa:

Isolation level xác định mức độ ảnh hưởng giữa các transaction đồng thời.

Mức độ Mô tả Hiệu suất An toàn
TRANSACTION_NONE Không hỗ trợ transaction - -
TRANSACTION_READ_UNCOMMITTED Có thể đọc dữ liệu chưa commit Cao nhất Thấp nhất
TRANSACTION_READ_COMMITTED Chỉ đọc dữ liệu đã commit Cao Trung bình
TRANSACTION_REPEATABLE_READ Đọc lặp lại được Trung bình Cao
TRANSACTION_SERIALIZABLE Các transaction chạy tuần tự Thấp nhất Cao nhất
Java
// Thiết lập isolation level
conn.setTransactionIsolation(Connection.TRANSACTION_READ_COMMITTED);

// Kiểm tra isolation level hiện tại
int level = conn.getTransactionIsolation();
System.out.println("Isolation level: " + level);

PHẦN 2: BATCH PROCESSING

2.1 BATCH PROCESSING LÀ GÌ?

Định nghĩa:

Batch Processing là kỹ thuật gom nhiều câu lệnh SQL thành một lô (batch) và gửi đến database một lần duy nhất.

Mục đích:

  • Giảm số lần giao tiếp giữa Java và Database
  • Tăng hiệu suất khi thực hiện nhiều câu lệnh
  • Đặc biệt hữu ích khi insert/update nhiều bản ghi

So sánh hiệu suất:

Không dùng batch: 
   Gửi 1000 lần → 1000 lần giao tiếp

Dùng batch: 
   Gom 1000 câu lệnh → Gửi 1 lần → Nhanh hơn nhiều

2.2 CÁC PHƯƠNG THỨC BATCH TRONG JDBC

Phương thức Mô tả
addBatch(String sql) Thêm câu lệnh SQL vào batch (Statement)
addBatch() Thêm câu lệnh đã set tham số vào batch (PreparedStatement)
executeBatch() Thực thi toàn bộ batch, trả về mảng kết quả
clearBatch() Xóa các câu lệnh trong batch

2.3 BATCH VỚI STATEMENT

BatchWithStatement.java
import java.sql.*;

public class BatchWithStatement {
    public static void main(String[] args) {
        Connection conn = null;
        Statement stmt = null;
        
        try {
            conn = DatabaseUtils.getConnection();
            conn.setAutoCommit(false); // Tắt auto-commit
            
            stmt = conn.createStatement();
            
            // Thêm các câu lệnh vào batch
            stmt.addBatch("INSERT INTO SanPham (TenSP, DonGia, SoLuongTon) VALUES ('Sản phẩm 1', 100000, 10)");
            stmt.addBatch("INSERT INTO SanPham (TenSP, DonGia, SoLuongTon) VALUES ('Sản phẩm 2', 200000, 20)");
            stmt.addBatch("INSERT INTO SanPham (TenSP, DonGia, SoLuongTon) VALUES ('Sản phẩm 3', 300000, 30)");
            stmt.addBatch("UPDATE SanPham SET SoLuongTon = SoLuongTon + 10 WHERE MaSP = 1");
            
            // Thực thi batch
            int[] results = stmt.executeBatch();
            
            // Kiểm tra kết quả
            for (int i = 0; i < results.length; i++) {
                System.out.println("Lệnh " + (i+1) + ": " + results[i] + " hàng bị ảnh hưởng");
            }
            
            // Commit nếu tất cả thành công
            conn.commit();
            System.out.println("Batch thực thi thành công!");
            
        } catch (SQLException e) {
            try {
                if (conn != null) conn.rollback();
                System.out.println("Đã rollback do lỗi!");
            } catch (SQLException ex) {
                ex.printStackTrace();
            }
            e.printStackTrace();
            
        } finally {
            DatabaseUtils.close(stmt);
            DatabaseUtils.close(conn);
        }
    }
}

2.4 BATCH VỚI PREPAREDSTATEMENT (KHUYẾN NGHỊ)

Ví dụ: Thêm nhiều sản phẩm cùng lúc

BatchWithPreparedStatement.java
import java.sql.*;
import java.util.ArrayList;
import java.util.List;

class SanPham {
    private String tenSP;
    private double donGia;
    private int soLuongTon;
    
    public SanPham(String tenSP, double donGia, int soLuongTon) {
        this.tenSP = tenSP;
        this.donGia = donGia;
        this.soLuongTon = soLuongTon;
    }
    
    // Getters
    public String getTenSP() { return tenSP; }
    public double getDonGia() { return donGia; }
    public int getSoLuongTon() { return soLuongTon; }
}

public class BatchWithPreparedStatement {
    public static void main(String[] args) {
        List<SanPham> products = new ArrayList<>();
        products.add(new SanPham("Laptop Asus", 18000000, 15));
        products.add(new SanPham("MacBook Air", 22000000, 10));
        products.add(new SanPham("iPad Pro", 20000000, 8));
        products.add(new SanPham("Apple Watch", 9000000, 12));
        products.add(new SanPham("AirPods", 3500000, 25));
        
        insertProducts(products);
    }
    
    public static void insertProducts(List<SanPham> products) {
        String sql = "INSERT INTO SanPham (TenSP, DonGia, SoLuongTon) VALUES (?, ?, ?)";
        Connection conn = null;
        PreparedStatement pstmt = null;
        
        try {
            conn = DatabaseUtils.getConnection();
            conn.setAutoCommit(false);
            
            pstmt = conn.prepareStatement(sql);
            
            // Thêm từng sản phẩm vào batch
            for (SanPham sp : products) {
                pstmt.setString(1, sp.getTenSP());
                pstmt.setDouble(2, sp.getDonGia());
                pstmt.setInt(3, sp.getSoLuongTon());
                pstmt.addBatch(); // Thêm vào batch
            }
            
            // Thực thi batch
            long startTime = System.currentTimeMillis();
            int[] results = pstmt.executeBatch();
            long endTime = System.currentTimeMillis();
            
            // Commit transaction
            conn.commit();
            
            // Kết quả
            int total = 0;
            for (int result : results) {
                total += result;
            }
            
            System.out.println("Đã thêm " + total + " sản phẩm thành công!");
            System.out.println("Thời gian: " + (endTime - startTime) + "ms");
            
        } catch (SQLException e) {
            try {
                if (conn != null) conn.rollback();
                System.out.println("Lỗi, đã rollback!");
            } catch (SQLException ex) {
                ex.printStackTrace();
            }
            e.printStackTrace();
            
        } finally {
            DatabaseUtils.close(pstmt);
            DatabaseUtils.close(conn);
        }
    }
}

2.5 XỬ LÝ LỖI TRONG BATCH

Khi thực thi batch, có thể xảy ra lỗi ở một số câu lệnh. Tùy vào cấu hình mà batch sẽ dừng lại hoặc tiếp tục.

BatchErrorHandling.java
public class BatchErrorHandling {
    public static void main(String[] args) {
        String sql = "INSERT INTO SanPham (TenSP, DonGia, SoLuongTon) VALUES (?, ?, ?)";
        
        try (Connection conn = DatabaseUtils.getConnection();
             PreparedStatement pstmt = conn.prepareStatement(sql)) {
            
            conn.setAutoCommit(false);
            
            // Câu lệnh 1: OK
            pstmt.setString(1, "Sản phẩm A");
            pstmt.setDouble(2, 100000);
            pstmt.setInt(3, 10);
            pstmt.addBatch();
            
            // Câu lệnh 2: Lỗi (tên sản phẩm quá dài? giá trị âm?)
            pstmt.setString(1, "Sản phẩm B với tên rất dài...");
            pstmt.setDouble(2, -50000); // Giá âm (có thể gây lỗi)
            pstmt.setInt(3, 5);
            pstmt.addBatch();
            
            // Câu lệnh 3: OK
            pstmt.setString(1, "Sản phẩm C");
            pstmt.setDouble(2, 200000);
            pstmt.setInt(3, 15);
            pstmt.addBatch();
            
            try {
                int[] results = pstmt.executeBatch();
                conn.commit();
                
                for (int i = 0; i < results.length; i++) {
                    System.out.println("Lệnh " + (i+1) + ": " + results[i]);
                }
                
            } catch (BatchUpdateException e) {
                // Lấy kết quả của các lệnh thành công trước khi lỗi
                int[] results = e.getUpdateCounts();
                
                for (int i = 0; i < results.length; i++) {
                    if (results[i] == Statement.EXECUTE_FAILED) {
                        System.out.println("Lệnh " + (i+1) + " thất bại!");
                    } else {
                        System.out.println("Lệnh " + (i+1) + ": " + results[i] + " hàng");
                    }
                }
                
                // Rollback hoặc xử lý tùy ý
                conn.rollback();
                System.out.println("Đã rollback toàn bộ batch");
            }
            
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }
}

PHẦN 3: CONNECTION POOLING

3.1 CONNECTION POOLING LÀ GÌ?

Định nghĩa:

Connection Pooling là kỹ thuật tái sử dụng các kết nối database. Thay vì tạo mới và đóng kết nối mỗi lần, ta tạo sẵn một pool các kết nối và lấy ra dùng khi cần.

Vấn đề khi không dùng Connection Pool:

  • Mỗi lần kết nối: phải tạo kết nối mới → tốn thời gian
  • Tài nguyên: mỗi kết nối chiếm bộ nhớ, CPU
  • Giới hạn: database giới hạn số kết nối đồng thời

Giải pháp Connection Pool:

┌─────────────────────┐
│   Connection Pool   │
├─────────────────────┤
│ [conn1] [conn2]     │
│ [conn3] [conn4]     │ ← Tạo sẵn N kết nối
└──────────┬──────────┘
           │ Lấy từ pool
    ┌──────▼──────┐
    │ Application │
    └──────┬──────┘
           │ Trả về pool
    ┌──────▼──────┐
    │   Pool      │ ← Kết nối được tái sử dụng
    └─────────────┘

3.2 CÁC THƯ VIỆN CONNECTION POOL PHỔ BIẾN

Thư viện Mô tả
HikariCP Nhanh nhất, nhẹ nhất, được khuyên dùng
Apache DBCP Của Apache, ổn định
C3P0 Cũ, chậm hơn
Tomcat JDBC Pool Dùng trong Tomcat

Trong khóa học này, chúng ta sẽ dùng HikariCP.

ConnectionPool.java
import com.zaxxer.hikari.HikariConfig;
import com.zaxxer.hikari.HikariDataSource;
import java.sql.*;
import java.io.InputStream;
import java.util.Properties;

public class ConnectionPool {
    private static HikariDataSource dataSource;
    
    static {
        try {
            // Đọc cấu hình từ file properties
            Properties props = new Properties();
            try (InputStream input = ConnectionPool.class.getClassLoader()
                    .getResourceAsStream("database.properties")) {
                if (input == null) {
                    System.out.println("Không tìm thấy file database.properties, dùng cấu hình mặc định");
                    createDefaultConfig();
                } else {
                    props.load(input);
                    createConfigFromProps(props);
                }
            }
            
        } catch (Exception e) {
            e.printStackTrace();
            createDefaultConfig();
        }
    }
    
    private static void createDefaultConfig() {
        HikariConfig config = new HikariConfig();
        config.setJdbcUrl("jdbc:mysql://localhost:3306/QuanLyBanHang?useSSL=false&serverTimezone=UTC");
        config.setUsername("root");
        config.setPassword("123456");
        config.setDriverClassName("com.mysql.cj.jdbc.Driver");
        config.setMaximumPoolSize(10);
        config.setMinimumIdle(5);
        config.setConnectionTimeout(30000);
        dataSource = new HikariDataSource(config);
    }
    
    private static void createConfigFromProps(Properties props) {
        HikariConfig config = new HikariConfig();
        config.setJdbcUrl(props.getProperty("db.url"));
        config.setUsername(props.getProperty("db.username"));
        config.setPassword(props.getProperty("db.password"));
        config.setDriverClassName(props.getProperty("db.driver"));
        config.setMaximumPoolSize(Integer.parseInt(props.getProperty("db.maxPoolSize", "10")));
        config.setMinimumIdle(Integer.parseInt(props.getProperty("db.minIdle", "5")));
        config.setConnectionTimeout(Long.parseLong(props.getProperty("db.connectionTimeout", "30000")));
        dataSource = new HikariDataSource(config);
    }
    
    public static Connection getConnection() throws SQLException {
        return dataSource.getConnection();
    }
    
    public static void close(Connection conn) {
        try { if (conn != null) conn.close(); } catch (SQLException e) {}
    }
    
    public static void close(Statement stmt) {
        try { if (stmt != null) stmt.close(); } catch (SQLException e) {}
    }
    
    public static void close(ResultSet rs) {
        try { if (rs != null) rs.close(); } catch (SQLException e) {}
    }
    
    public static void close(Connection conn, Statement stmt, ResultSet rs) {
        close(rs);
        close(stmt);
        close(conn);
    }
    
    public static void shutdown() {
        if (dataSource != null && !dataSource.isClosed()) {
            dataSource.close();
            System.out.println("Connection pool đã đóng");
        }
    }
}

PHẦN 4: DAO PATTERN

4.1 DAO PATTERN LÀ GÌ?

Định nghĩa:

DAO (Data Access Object) là một design pattern tách biệt logic truy cập dữ liệu khỏi logic nghiệp vụ.

Mục đích:

  • Tách biệt code database khỏi business logic
  • Dễ bảo trì, dễ thay đổi database
  • Tái sử dụng code
  • Dễ test

Cấu trúc DAO Pattern:

┌─────────────────┐
│    Business     │
│    Layer        │
└────────┬────────┘
         │ Sử dụng
┌────────▼────────┐
│   DAO Interface │ (Contract)
└────────┬────────┘
         │ Implement
┌────────▼────────┐
│   DAO Impl      │ (JDBC, Hibernate, JPA...)
└────────┬────────┘
         │
┌────────▼────────┐
│    Database     │
└─────────────────┘

4.2 CÁC THÀNH PHẦN CỦA DAO PATTERN

Thành phần Mô tả
DTO/Entity Đối tượng chứa dữ liệu (POJO)
DAO Interface Định nghĩa các phương thức CRUD
DAO Implementation Cài đặt cụ thể (JDBC, Hibernate,...)
DAO Factory Tạo đối tượng DAO phù hợp

Bước 3: Tạo DAO Implementation với JDBC

dao/impl/SanPhamDAOImpl.java
package dao.impl;

import dao.SanPhamDAO;
import entity.SanPham;
import utils.ConnectionPool;

import java.sql.*;
import java.util.ArrayList;
import java.util.List;

public class SanPhamDAOImpl implements SanPhamDAO {
    
    // SQL queries
    private static final String SELECT_ALL = "SELECT * FROM SanPham";
    private static final String SELECT_BY_ID = "SELECT * FROM SanPham WHERE MaSP = ?";
    private static final String INSERT = "INSERT INTO SanPham (TenSP, DonGia, SoLuongTon) VALUES (?, ?, ?)";
    private static final String UPDATE = "UPDATE SanPham SET TenSP = ?, DonGia = ?, SoLuongTon = ? WHERE MaSP = ?";
    private static final String DELETE = "DELETE FROM SanPham WHERE MaSP = ?";
    private static final String SEARCH_BY_NAME = "SELECT * FROM SanPham WHERE TenSP LIKE ?";
    private static final String SELECT_BY_PRICE = "SELECT * FROM SanPham WHERE DonGia BETWEEN ? AND ?";
    private static final String SELECT_LOW_STOCK = "SELECT * FROM SanPham WHERE SoLuongTon < ?";
    private static final String COUNT = "SELECT COUNT(*) FROM SanPham";
    
    @Override
    public SanPham getById(int id) {
        SanPham result = null;
        
        try (Connection conn = ConnectionPool.getConnection();
             PreparedStatement pstmt = conn.prepareStatement(SELECT_BY_ID)) {
            
            pstmt.setInt(1, id);
            ResultSet rs = pstmt.executeQuery();
            
            if (rs.next()) {
                result = mapResultSetToSanPham(rs);
            }
            
        } catch (SQLException e) {
            e.printStackTrace();
        }
        
        return result;
    }
    
    @Override
    public List<SanPham> getAll() {
        List<SanPham> list = new ArrayList<>();
        
        try (Connection conn = ConnectionPool.getConnection();
             Statement stmt = conn.createStatement();
             ResultSet rs = stmt.executeQuery(SELECT_ALL)) {
            
            while (rs.next()) {
                list.add(mapResultSetToSanPham(rs));
            }
            
        } catch (SQLException e) {
            e.printStackTrace();
        }
        
        return list;
    }
    
    @Override
    public boolean insert(SanPham sp) {
        try (Connection conn = ConnectionPool.getConnection();
             PreparedStatement pstmt = conn.prepareStatement(INSERT, Statement.RETURN_GENERATED_KEYS)) {
            
            pstmt.setString(1, sp.getTenSP());
            pstmt.setDouble(2, sp.getDonGia());
            pstmt.setInt(3, sp.getSoLuongTon());
            
            int affectedRows = pstmt.executeUpdate();
            
            if (affectedRows > 0) {
                // Lấy ID tự động sinh ra
                ResultSet rs = pstmt.getGeneratedKeys();
                if (rs.next()) {
                    sp.setMaSP(rs.getInt(1));
                }
                return true;
            }
            
        } catch (SQLException e) {
            e.printStackTrace();
        }
        
        return false;
    }
    
    @Override
    public boolean update(SanPham sp) {
        try (Connection conn = ConnectionPool.getConnection();
             PreparedStatement pstmt = conn.prepareStatement(UPDATE)) {
            
            pstmt.setString(1, sp.getTenSP());
            pstmt.setDouble(2, sp.getDonGia());
            pstmt.setInt(3, sp.getSoLuongTon());
            pstmt.setInt(4, sp.getMaSP());
            
            int affectedRows = pstmt.executeUpdate();
            return affectedRows > 0;
            
        } catch (SQLException e) {
            e.printStackTrace();
            return false;
        }
    }
    
    @Override
    public boolean delete(int id) {
        try (Connection conn = ConnectionPool.getConnection();
             PreparedStatement pstmt = conn.prepareStatement(DELETE)) {
            
            pstmt.setInt(1, id);
            int affectedRows = pstmt.executeUpdate();
            return affectedRows > 0;
            
        } catch (SQLException e) {
            e.printStackTrace();
            return false;
        }
    }
    
    @Override
    public List<SanPham> searchByName(String keyword) {
        List<SanPham> list = new ArrayList<>();
        
        try (Connection conn = ConnectionPool.getConnection();
             PreparedStatement pstmt = conn.prepareStatement(SEARCH_BY_NAME)) {
            
            pstmt.setString(1, "%" + keyword + "%");
            ResultSet rs = pstmt.executeQuery();
            
            while (rs.next()) {
                list.add(mapResultSetToSanPham(rs));
            }
            
        } catch (SQLException e) {
            e.printStackTrace();
        }
        
        return list;
    }
    
    @Override
    public List<SanPham> getByPriceRange(double min, double max) {
        List<SanPham> list = new ArrayList<>();
        
        try (Connection conn = ConnectionPool.getConnection();
             PreparedStatement pstmt = conn.prepareStatement(SELECT_BY_PRICE)) {
            
            pstmt.setDouble(1, min);
            pstmt.setDouble(2, max);
            ResultSet rs = pstmt.executeQuery();
            
            while (rs.next()) {
                list.add(mapResultSetToSanPham(rs));
            }
            
        } catch (SQLException e) {
            e.printStackTrace();
        }
        
        return list;
    }
    
    @Override
    public List<SanPham> getLowStock(int threshold) {
        List<SanPham> list = new ArrayList<>();
        
        try (Connection conn = ConnectionPool.getConnection();
             PreparedStatement pstmt = conn.prepareStatement(SELECT_LOW_STOCK)) {
            
            pstmt.setInt(1, threshold);
            ResultSet rs = pstmt.executeQuery();
            
            while (rs.next()) {
                list.add(mapResultSetToSanPham(rs));
            }
            
        } catch (SQLException e) {
            e.printStackTrace();
        }
        
        return list;
    }
    
    @Override
    public int count() {
        try (Connection conn = ConnectionPool.getConnection();
             Statement stmt = conn.createStatement();
             ResultSet rs = stmt.executeQuery(COUNT)) {
            
            if (rs.next()) {
                return rs.getInt(1);
            }
            
        } catch (SQLException e) {
            e.printStackTrace();
        }
        
        return 0;
    }
    
    // Helper method: map ResultSet → SanPham object
    private SanPham mapResultSetToSanPham(ResultSet rs) throws SQLException {
        SanPham sp = new SanPham();
        sp.setMaSP(rs.getInt("MaSP"));
        sp.setTenSP(rs.getString("TenSP"));
        sp.setDonGia(rs.getDouble("DonGia"));
        sp.setSoLuongTon(rs.getInt("SoLuongTon"));
        return sp;
    }
}

Bước 5: Sử dụng DAO trong Business Layer

service/SanPhamService.java
package service;

import dao.SanPhamDAO;
import entity.SanPham;
import factory.DAOFactory;
import java.util.List;

public class SanPhamService {
    
    private SanPhamDAO sanPhamDAO;
    
    public SanPhamService() {
        this.sanPhamDAO = DAOFactory.getSanPhamDAO();
    }
    
    // Business logic methods
    
    public List<SanPham> getAllProducts() {
        return sanPhamDAO.getAll();
    }
    
    public SanPham getProductById(int id) {
        if (id <= 0) {
            throw new IllegalArgumentException("ID không hợp lệ");
        }
        return sanPhamDAO.getById(id);
    }
    
    public boolean addProduct(SanPham sp) {
        // Validate business rules
        if (sp.getTenSP() == null || sp.getTenSP().trim().isEmpty()) {
            throw new IllegalArgumentException("Tên sản phẩm không được để trống");
        }
        if (sp.getDonGia() < 0) {
            throw new IllegalArgumentException("Đơn giá không thể âm");
        }
        if (sp.getSoLuongTon() < 0) {
            throw new IllegalArgumentException("Số lượng tồn không thể âm");
        }
        
        return sanPhamDAO.insert(sp);
    }
    
    public boolean updateProduct(SanPham sp) {
        if (sp.getMaSP() <= 0) {
            throw new IllegalArgumentException("Mã sản phẩm không hợp lệ");
        }
        return sanPhamDAO.update(sp);
    }
    
    public boolean deleteProduct(int id) {
        if (id <= 0) {
            throw new IllegalArgumentException("ID không hợp lệ");
        }
        return sanPhamDAO.delete(id);
    }
    
    public List<SanPham> searchProducts(String keyword) {
        if (keyword == null || keyword.trim().isEmpty()) {
            return getAllProducts();
        }
        return sanPhamDAO.searchByName(keyword);
    }
    
    public List<SanPham> getProductsByPriceRange(double min, double max) {
        if (min > max) {
            throw new IllegalArgumentException("Khoảng giá không hợp lệ");
        }
        return sanPhamDAO.getByPriceRange(min, max);
    }
    
    public List<SanPham> getLowStockProducts(int threshold) {
        return sanPhamDAO.getLowStock(threshold);
    }
    
    public int getTotalProducts() {
        return sanPhamDAO.count();
    }
}

            System.out.println("Thêm thành công! Mã SP: " + newProduct.getMaSP());
        }
        
        // 3. Tìm kiếm
        System.out.println("\n=== TÌM SẢN PHẨM CÓ TÊN CHỨA 'Laptop' ===");
        List<SanPham> search = service.searchProducts("Laptop");
        search.forEach(System.out::println);
        
        // 4. Lọc theo giá
        System.out.println("\n=== SẢN PHẨM GIÁ 5TR - 15TR ===");
        List<SanPham> priceRange = service.getProductsByPriceRange(5000000, 15000000);
        priceRange.forEach(System.out::println);
        
        // 5. Sản phẩm sắp hết hàng
        System.out.println("\n=== SẢN PHẨM TỒN KHO < 10 ===");
        List<SanPham> lowStock = service.getLowStockProducts(10);
        lowStock.forEach(System.out::println);
        
        // 6. Đếm tổng số
        System.out.println("\nTổng số sản phẩm: " + service.getTotalProducts());
        
        // 7. Cập nhật
        System.out.println("\n=== CẬP NHẬT SẢN PHẨM MÃ 1 ===");
        SanPham sp1 = service.getProductById(1);
        if (sp1 != null) {
            sp1.setSoLuongTon(sp1.getSoLuongTon() + 5);
            if (service.updateProduct(sp1)) {
                System.out.println("Cập nhật thành công: " + sp1);
            }
        }
        
        // Đóng connection pool
        ConnectionPool.shutdown();
    }
}
