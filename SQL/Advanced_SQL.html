Code tham khảo

Tạo database và bảng
-- Tạo database
CREATE DATABASE QuanLyBanHang;
USE QuanLyBanHang;

-- Bảng KhachHang
CREATE TABLE KhachHang (
    MaKH INT PRIMARY KEY,
    TenKH VARCHAR(100) NOT NULL,
    DiaChi VARCHAR(200),
    DienThoai VARCHAR(15),
    Email VARCHAR(100)
);

-- Bảng SanPham
CREATE TABLE SanPham (
    MaSP INT PRIMARY KEY,
    TenSP VARCHAR(100) NOT NULL,
    DonGia DECIMAL(10,2),
    SoLuongTon INT
);

-- Bảng HoaDon
CREATE TABLE HoaDon (
    MaHD INT PRIMARY KEY,
    MaKH INT,
    NgayLap DATE,
    TongTien DECIMAL(12,2),
    FOREIGN KEY (MaKH) REFERENCES KhachHang(MaKH)
);

-- Bảng ChiTietHD
CREATE TABLE ChiTietHD (
    MaHD INT,
    MaSP INT,
    SoLuong INT,
    DonGia DECIMAL(10,2),
    PRIMARY KEY (MaHD, MaSP),
    FOREIGN KEY (MaHD) REFERENCES HoaDon(MaHD),
    FOREIGN KEY (MaSP) REFERENCES SanPham(MaSP)
);

Yêu cầu

Viết câu lệnh SQL thực hiện:

  • a) Hiển thị danh sách các hóa đơn gồm: MaHD, TenKH, NgayLap, TongTien
  • b) Hiển thị tổng số lượng sản phẩm mà mỗi khách hàng đã mua, gồm: MaKH, TenKH, TongSoLuong
  • c) Hiển thị khách hàng có tổng tiền mua hàng lớn hơn 30.000.000, gồm: MaKH, TenKH, TongTienDaMua
Đáp án tham khảo
-- a) Hiển thị danh sách các hóa đơn
SELECT
    hd.MaHD,
    kh.TenKH,
    hd.NgayLap,
    hd.TongTien
FROM HoaDon hd
INNER JOIN KhachHang kh ON hd.MaKH = kh.MaKH;

-- b) Tổng số lượng sản phẩm mỗi khách hàng đã mua
SELECT
    kh.MaKH,
    kh.TenKH,
    SUM(cthd.SoLuong) AS TongSoLuong
FROM KhachHang kh
LEFT JOIN HoaDon hd ON kh.MaKH = hd.MaKH
LEFT JOIN ChiTietHD cthd ON hd.MaHD = cthd.MaHD
GROUP BY kh.MaKH, kh.TenKH;

-- c) Khách hàng có tổng tiền mua hàng > 30.000.000
SELECT
    kh.MaKH,
    kh.TenKH,
    SUM(hd.TongTien) AS TongTienDaMua
FROM KhachHang kh
INNER JOIN HoaDon hd ON kh.MaKH = hd.MaKH
GROUP BY kh.MaKH, kh.TenKH
HAVING SUM(hd.TongTien) > 30000000;

PHẦN 1: AGGREGATE FUNCTIONS (HÀM TỔNG HỢP)

1.1 TỔNG QUAN VỀ AGGREGATE FUNCTIONS

Định nghĩa:

Aggregate Functions là các hàm thực hiện tính toán trên một tập hợp các hàng và trả về một giá trị duy nhất.

Mục đích:

  • Tính tổng, trung bình, đếm số lượng
  • Tìm giá trị lớn nhất, nhỏ nhất
  • Thống kê dữ liệu

Đặc điểm:

  • Thường kết hợp với GROUP BY để nhóm dữ liệu
  • Bỏ qua các giá trị NULL (trừ COUNT(*))
  • Trả về một hàng duy nhất cho mỗi nhóm

Các hàm aggregate cơ bản:

Hàm Mô tả Kiểu trả về
COUNT() Đếm số lượng INT
SUM() Tính tổng Số
AVG() Tính trung bình Số
MAX() Tìm giá trị lớn nhất Cùng kiểu dữ liệu
MIN() Tìm giá trị nhỏ nhất Cùng kiểu dữ liệu

1.2 COUNT() - ĐẾM SỐ LƯỢNG

Mục đích:

Đếm số lượng hàng hoặc số lượng giá trị không NULL trong một cột.

Cú pháp:

Cú pháp
COUNT(*) -- Đếm tất cả hàng (kể cả NULL)
COUNT(column) -- Đếm các giá trị không NULL của cột
COUNT(DISTINCT column) -- Đếm các giá trị duy nhất

Ví dụ 1: COUNT cơ bản

SQL
-- Đếm tổng số khách hàng
SELECT COUNT(*) AS SoKhachHang
FROM KhachHang;
-- Kết quả: 5

-- Đếm số khách hàng có email
SELECT COUNT(Email) AS CoEmail
FROM KhachHang;
-- Kết quả: 5 (tất cả đều có email)

Ví dụ 2: COUNT với DISTINCT

SQL
-- Đếm số lượng khách hàng đã mua hàng
SELECT COUNT(DISTINCT MaKH) AS KhachHangMuaHang
FROM HoaDon;
-- Kết quả: 4 (khách hàng 1,2,3,4 - khách 5 chưa mua)

-- Đếm số lượng sản phẩm đã được bán
SELECT COUNT(DISTINCT MaSP) AS SanPhamDaBan
FROM ChiTietHD;
-- Kết quả: 5 (tất cả sản phẩm đều đã được bán)

Ví dụ 3: COUNT với điều kiện

SQL
-- Đếm số hóa đơn trong tháng 1/2024
SELECT COUNT(*) AS HoaDonThang1
FROM HoaDon
WHERE NgayLap BETWEEN '2024-01-01' AND '2024-01-31';
-- Kết quả: 2

-- Đếm số sản phẩm có giá > 10 triệu
SELECT COUNT(*) AS SanPhamDatien
FROM SanPham
WHERE DonGia > 10000000;
-- Kết quả: 2 (Laptop và iPhone)

1.3 SUM() - TÍNH TỔNG

Mục đích:

Tính tổng các giá trị của một cột (chỉ áp dụng cho kiểu số).

Cú pháp:

Cú pháp
SUM(column)
SUM(DISTINCT column) -- Tính tổng các giá trị duy nhất

Ví dụ 1: SUM cơ bản

SQL
-- Tính tổng doanh thu từ tất cả hóa đơn
SELECT SUM(TongTien) AS TongDoanhThu
FROM HoaDon;
-- Kết quả: 112300000 (45tr + 27tr + 30tr + 2.3tr + 8tr)

-- Tính tổng số lượng tồn kho
SELECT SUM(SoLuongTon) AS TongTonKho
FROM SanPham;
-- Kết quả: 380 (50+30+100+80+120)

Ví dụ 2: SUM với điều kiện

SQL
-- Tính tổng doanh thu trong tháng 1/2024
SELECT SUM(TongTien) AS DoanhThuThang1
FROM HoaDon
WHERE NgayLap BETWEEN '2024-01-01' AND '2024-01-31';
-- Kết quả: 72000000 (45tr + 27tr)

-- Tính tổng doanh thu của khách hàng có mã 1
SELECT SUM(TongTien) AS TongTienKH1
FROM HoaDon
WHERE MaKH = 1;
-- Kết quả: 75000000 (45tr + 30tr)

1.4 AVG() - TÍNH TRUNG BÌNH

Mục đích:

Tính giá trị trung bình của một cột (chỉ áp dụng cho kiểu số).

Cú pháp:

Cú pháp
AVG(column)
AVG(DISTINCT column) -- Trung bình của các giá trị duy nhất

Ví dụ 1: AVG cơ bản

SQL
-- Tính giá trung bình của các sản phẩm
SELECT AVG(DonGia) AS GiaTrungBinh
FROM SanPham;
-- Kết quả: 8760000 (tổng 43.8tr / 5 sản phẩm)

-- Tính số lượng trung bình mỗi lần mua
SELECT AVG(SoLuong) AS SoLuongTB
FROM ChiTietHD;
-- Kết quả: 1.9 (tổng 19 / 10 dòng)

Ví dụ 2: AVG với điều kiện

SQL
-- Tính giá trung bình của sản phẩm > 5 triệu
SELECT AVG(DonGia) AS GiaTBCao
FROM SanPham
WHERE DonGia > 5000000;
-- Kết quả: 20000000 (chỉ Laptop và iPhone)

-- Tính số lượng trung bình của sản phẩm có mã 1 trong hóa đơn
SELECT AVG(SoLuong) AS SoLuongTB_SP1
FROM ChiTietHD
WHERE MaSP = 1;
-- Kết quả: 1.5 (2 và 1)

1.5 MAX() và MIN() - LỚN NHẤT VÀ NHỎ NHẤT

Mục đích:

Tìm giá trị lớn nhất và nhỏ nhất trong một cột.

Cú pháp:

Cú pháp
MAX(column)
MIN(column)

Ví dụ 1: MAX và MIN với số

SQL
-- Tìm sản phẩm đắt nhất và rẻ nhất
SELECT
    MAX(DonGia) AS GiaDatNhat,
    MIN(DonGia) AS GiaReNhat
FROM SanPham;
-- Kết quả: GiaDatNhat = 25000000, GiaReNhat = 800000

-- Tìm hóa đơn có tổng tiền lớn nhất và nhỏ nhất
SELECT
    MAX(TongTien) AS HoaDonLonNhat,
    MIN(TongTien) AS HoaDonNhoNhat
FROM HoaDon;
-- Kết quả: 45000000 và 2300000

Ví dụ 2: MAX và MIN với ngày tháng

SQL
-- Tìm ngày lập hóa đơn đầu tiên và gần nhất
SELECT
    MIN(NgayLap) AS HoaDonDauTien,
    MAX(NgayLap) AS HoaDonGanNhat
FROM HoaDon;
-- Kết quả: 2024-01-15 và 2024-03-01

-- Tìm khách hàng có tên đầu và cuối trong bảng chữ cái
SELECT
    MIN(TenKH) AS TenDauTien,
    MAX(TenKH) AS TenCuoiCung
FROM KhachHang;
-- Kết quả: Hoàng Văn Em và Trần Thị Bích (theo alphabet)

1.6 GROUP BY - NHÓM DỮ LIỆU

Mục đích:

Nhóm các hàng có cùng giá trị thành một nhóm để thực hiện các hàm aggregate.

Cú pháp:

Cú pháp
SELECT column1, aggregate_function(column2)
FROM table_name
GROUP BY column1;

Ví dụ 1: GROUP BY cơ bản

SQL
-- Tính tổng tiền mỗi khách hàng đã chi tiêu
SELECT
    MaKH,
    SUM(TongTien) AS TongChiTieu
FROM HoaDon
GROUP BY MaKH;

-- Kết quả:
-- MaKH = 1: 75000000
-- MaKH = 2: 27000000
-- MaKH = 3: 2300000
-- MaKH = 4: 8000000
-- MaKH = 5: NULL (không có hóa đơn)

Ví dụ 2: GROUP BY với nhiều cột

SQL
-- Tính tổng số lượng bán của mỗi sản phẩm
SELECT
    MaSP,
    SUM(SoLuong) AS TongSoLuong
FROM ChiTietHD
GROUP BY MaSP
ORDER BY MaSP;

-- Kết quả:
-- MaSP 1: 3 (2+1)
-- MaSP 2: 1
-- MaSP 3: 4 (3+1)
-- MaSP 4: 4 (2+2)
-- MaSP 5: 8 (2+1+5)

Ví dụ 3: GROUP BY với JOIN

SQL
-- Tính tổng tiền theo từng khách hàng (có tên)
SELECT
    kh.TenKH,
    SUM(hd.TongTien) AS TongChiTieu
FROM KhachHang kh
LEFT JOIN HoaDon hd ON kh.MaKH = hd.MaKH
GROUP BY kh.MaKH, kh.TenKH
ORDER BY TongChiTieu DESC;

-- Kết quả:
-- Nguyễn Văn An: 75000000
-- Trần Thị Bích: 27000000
-- Phạm Thị Dung: 8000000
-- Lê Văn Cường: 2300000
-- Hoàng Văn Em: NULL

1.7 HAVING - LỌC NHÓM

Mục đích:

Lọc các nhóm sau khi đã GROUP BY (giống WHERE nhưng dùng cho nhóm).

Cú pháp:

Cú pháp
SELECT column1, aggregate_function(column2)
FROM table_name
GROUP BY column1
HAVING condition;

Ví dụ 1: HAVING cơ bản

SQL
-- Lấy các khách hàng có tổng chi tiêu > 10 triệu
SELECT
    MaKH,
    SUM(TongTien) AS TongChiTieu
FROM HoaDon
GROUP BY MaKH
HAVING SUM(TongTien) > 10000000;

-- Kết quả: khách hàng 1 (75tr) và 2 (27tr)

-- Không thể dùng WHERE vì WHERE không dùng được với aggregate
-- WHERE SUM(TongTien) > 10000000 là SAI

Ví dụ 2: HAVING với điều kiện phức tạp

SQL
-- Lấy các sản phẩm có tổng số lượng bán > 3
SELECT
    MaSP,
    SUM(SoLuong) AS TongSoLuong
FROM ChiTietHD
GROUP BY MaSP
HAVING SUM(SoLuong) > 3;

-- Kết quả:
-- MaSP 3: 4
-- MaSP 4: 4
-- MaSP 5: 8

Ví dụ 3: WHERE và HAVING kết hợp

SQL
-- Lấy các khách hàng ở Hà Nội có tổng chi tiêu > 10 triệu
SELECT
    kh.TenKH,
    kh.DiaChi,
    SUM(hd.TongTien) AS TongChiTieu
FROM KhachHang kh
LEFT JOIN HoaDon hd ON kh.MaKH = hd.MaKH
WHERE kh.DiaChi = 'Hà Nội' -- Lọc trước khi nhóm
GROUP BY kh.MaKH, kh.TenKH, kh.DiaChi
HAVING SUM(hd.TongTien) > 10000000; -- Lọc sau khi nhóm

-- Kết quả: Nguyễn Văn An - 75tr

PHẦN 2: JOINS

2.1 TỔNG QUAN VỀ JOINS

Định nghĩa:

JOIN là câu lệnh dùng để kết hợp dữ liệu từ hai hoặc nhiều bảng dựa trên mối quan hệ giữa các cột.

Mục đích:

  • Lấy dữ liệu từ nhiều bảng cùng lúc
  • Kết hợp thông tin có liên quan
  • Tránh lặp lại dữ liệu

Các loại JOIN:

Loại JOIN Mô tả
INNER JOIN Lấy các hàng có giá trị khớp ở cả hai bảng
LEFT JOIN Lấy tất cả từ bảng trái, khớp với bảng phải (NULL nếu không khớp)
RIGHT JOIN Lấy tất cả từ bảng phải, khớp với bảng trái (NULL nếu không khớp)
FULL JOIN Lấy tất cả từ cả hai bảng (NULL nếu không khớp)
CROSS JOIN Tích Descartes - mỗi hàng bảng A ghép với mỗi hàng bảng B

2.2 INNER JOIN

Mục đích:

Chỉ lấy các hàng có giá trị khớp ở cả hai bảng.

Cú pháp:

Cú pháp
SELECT columns
FROM table1
INNER JOIN table2 ON table1.column = table2.column;

Ví dụ 1: INNER JOIN cơ bản

SQL
-- Lấy thông tin hóa đơn kèm tên khách hàng
SELECT
    hd.MaHD,
    hd.NgayLap,
    kh.TenKH,
    hd.TongTien
FROM HoaDon hd
INNER JOIN KhachHang kh ON hd.MaKH = kh.MaKH;

-- Kết quả: 5 hóa đơn (khách 5 không có hóa đơn nên không xuất hiện)

Ví dụ 2: INNER JOIN nhiều bảng

SQL
-- Lấy chi tiết hóa đơn kèm tên sản phẩm và tên khách hàng
SELECT
    hd.MaHD,
    kh.TenKH,
    sp.TenSP,
    cthd.SoLuong,
    cthd.DonGia,
    (cthd.SoLuong * cthd.DonGia) AS ThanhTien
FROM HoaDon hd
INNER JOIN KhachHang kh ON hd.MaKH = kh.MaKH
INNER JOIN ChiTietHD cthd ON hd.MaHD = cthd.MaHD
INNER JOIN SanPham sp ON cthd.MaSP = sp.MaSP
ORDER BY hd.MaHD, sp.TenSP;

Ví dụ 3: INNER JOIN với điều kiện

SQL
-- Lấy hóa đơn của khách hàng ở Hà Nội
SELECT
    hd.MaHD,
    kh.TenKH,
    kh.DiaChi,
    hd.TongTien
FROM HoaDon hd
INNER JOIN KhachHang kh ON hd.MaKH = kh.MaKH
WHERE kh.DiaChi = 'Hà Nội';
-- Kết quả: hóa đơn của Nguyễn Văn An

2.3 LEFT JOIN (LEFT OUTER JOIN)

Mục đích:

Lấy tất cả hàng từ bảng trái (bảng đầu tiên) và các hàng khớp từ bảng phải. Nếu không khớp, các cột từ bảng phải sẽ là NULL.

Cú pháp:

Cú pháp
SELECT columns
FROM table1
LEFT JOIN table2 ON table1.column = table2.column;

Ví dụ 1: LEFT JOIN cơ bản

SQL
-- Lấy tất cả khách hàng và hóa đơn (nếu có)
SELECT
    kh.MaKH,
    kh.TenKH,
    hd.MaHD,
    hd.TongTien
FROM KhachHang kh
LEFT JOIN HoaDon hd ON kh.MaKH = hd.MaKH;

-- Kết quả:
-- KH 1-4 có hóa đơn, KH 5 không có hóa đơn (các cột hóa đơn NULL)

Ví dụ 2: LEFT JOIN để tìm dữ liệu không khớp

SQL
-- Tìm khách hàng chưa từng mua hàng
SELECT
    kh.MaKH,
    kh.TenKH,
    kh.DiaChi
FROM KhachHang kh
LEFT JOIN HoaDon hd ON kh.MaKH = hd.MaKH
WHERE hd.MaHD IS NULL;

-- Kết quả: Hoàng Văn Em (chưa có hóa đơn)

Ví dụ 3: LEFT JOIN với aggregate

SQL
-- Tính tổng chi tiêu của mỗi khách hàng (kể cả chưa mua)
SELECT
    kh.MaKH,
    kh.TenKH,
    SUM(hd.TongTien) AS TongChiTieu
FROM KhachHang kh
LEFT JOIN HoaDon hd ON kh.MaKH = hd.MaKH
GROUP BY kh.MaKH, kh.TenKH
ORDER BY TongChiTieu DESC;

-- Kết quả: KH 5 có tổng = NULL (có thể dùng COALESCE để thay bằng 0)

2.4 RIGHT JOIN (RIGHT OUTER JOIN)

Mục đích:

Lấy tất cả hàng từ bảng phải (bảng thứ hai) và các hàng khớp từ bảng trái. Ngược lại với LEFT JOIN.

Cú pháp:

Cú pháp
SELECT columns
FROM table1
RIGHT JOIN table2 ON table1.column = table2.column;

Ví dụ:

SQL
-- Lấy tất cả hóa đơn và thông tin khách hàng (tương đương LEFT JOIN)
SELECT
    kh.MaKH,
    kh.TenKH,
    hd.MaHD,
    hd.TongTien
FROM KhachHang kh
RIGHT JOIN HoaDon hd ON kh.MaKH = hd.MaKH;

-- Kết quả: giống INNER JOIN vì mọi hóa đơn đều có khách hàng

Lưu ý: RIGHT JOIN ít dùng hơn LEFT JOIN vì có thể đảo ngược thứ tự bảng.

2.5 FULL JOIN (FULL OUTER JOIN)

Mục đích:

Lấy tất cả hàng từ cả hai bảng, điền NULL nếu không khớp.

Cú pháp:

Cú pháp
SELECT columns
FROM table1
FULL JOIN table2 ON table1.column = table2.column;

Ví dụ:

SQL
-- Lấy tất cả khách hàng và tất cả hóa đơn
SELECT
    kh.MaKH,
    kh.TenKH,
    hd.MaHD,
    hd.TongTien
FROM KhachHang kh
FULL JOIN HoaDon hd ON kh.MaKH = hd.MaKH;

-- Kết quả (MySQL không hỗ trợ FULL JOIN trực tiếp, phải dùng UNION)

Lưu ý: MySQL không hỗ trợ FULL JOIN, có thể thay thế bằng UNION của LEFT JOIN và RIGHT JOIN.

2.6 CROSS JOIN

Mục đích:

Tạo tích Descartes - mỗi hàng từ bảng A kết hợp với mỗi hàng từ bảng B.

Cú pháp:

Cú pháp
SELECT columns
FROM table1
CROSS JOIN table2;

Ví dụ:

SQL
-- Tạo tất cả cặp (khách hàng, sản phẩm) có thể
SELECT
    kh.TenKH,
    sp.TenSP
FROM KhachHang kh
CROSS JOIN SanPham sp
ORDER BY kh.TenKH, sp.TenSP;

-- Kết quả: 5 khách * 5 sản phẩm = 25 dòng

2.7 SELF JOIN

Mục đích:

JOIN một bảng với chính nó, dùng alias để phân biệt.

Cú pháp:

Cú pháp
SELECT columns
FROM table t1
JOIN table t2 ON t1.column = t2.column;

Ví dụ:

SQL
-- Giả sử bảng nhân viên có cấu trúc: MaNV, TenNV, MaNQL (mã quản lý)
-- Tìm tên nhân viên và tên quản lý của họ
SELECT
    nv.TenNV AS NhanVien,
    ql.TenNV AS QuanLy
FROM NhanVien nv
LEFT JOIN NhanVien ql ON nv.MaNQL = ql.MaNV;

PHẦN 3: SUBQUERIES

3.1 TỔNG QUAN VỀ SUBQUERIES

Định nghĩa:

Subquery (truy vấn con) là một câu lệnh SELECT nằm bên trong một câu lệnh SELECT, INSERT, UPDATE, DELETE khác.

Mục đích:

  • Lấy dữ liệu để so sánh với kết quả truy vấn khác
  • Tính toán giá trị trung gian
  • Thay thế cho JOIN trong một số trường hợp

Các loại subquery:

Loại Mô tả
Scalar Subquery Trả về một giá trị duy nhất
Row Subquery Trả về một hàng
Table Subquery Trả về một bảng
Correlated Subquery Tham chiếu đến bảng bên ngoài

3.2 SCALAR SUBQUERY

Mục đích:

Subquery trả về một giá trị duy nhất (một cột, một hàng).

Cú pháp:

Cú pháp
SELECT column1, (SELECT ...) AS alias
FROM table;

Ví dụ 1: Scalar Subquery trong SELECT

SQL
-- Lấy thông tin hóa đơn và tổng doanh thu trung bình
SELECT
    MaHD,
    TongTien,
    (SELECT AVG(TongTien) FROM HoaDon) AS TB_DoanhThu,
    TongTien - (SELECT AVG(TongTien) FROM HoaDon) AS ChenhLech
FROM HoaDon;

-- Kết quả: mỗi hóa đơn được so sánh với giá trị trung bình

Ví dụ 2: Scalar Subquery trong WHERE

SQL
-- Lấy các hóa đơn có tổng tiền lớn hơn trung bình
SELECT MaHD, TongTien
FROM HoaDon
WHERE TongTien > (SELECT AVG(TongTien) FROM HoaDon);

-- Kết quả: các hóa đơn > 22.46tr

Ví dụ 3: Scalar Subquery trong HAVING

SQL
-- Lấy khách hàng có tổng chi tiêu lớn hơn trung bình
SELECT
    MaKH,
    SUM(TongTien) AS TongChiTieu
FROM HoaDon
GROUP BY MaKH
HAVING SUM(TongTien) > (SELECT AVG(TongTien) FROM HoaDon);

3.3 SUBQUERY VỚI IN

Mục đích:

Kiểm tra giá trị có nằm trong tập kết quả của subquery không.

Cú pháp:

Cú pháp
WHERE column IN (subquery)

Ví dụ 1: IN với subquery

SQL
-- Lấy thông tin khách hàng đã mua hàng
SELECT *
FROM KhachHang
WHERE MaKH IN (SELECT DISTINCT MaKH FROM HoaDon);

-- Kết quả: 4 khách hàng (trừ khách 5)

Ví dụ 2: NOT IN với subquery

SQL
-- Lấy khách hàng chưa mua hàng
SELECT *
FROM KhachHang
WHERE MaKH NOT IN (SELECT DISTINCT MaKH FROM HoaDon);

-- Kết quả: khách hàng 5

Ví dụ 3: IN với subquery nhiều cột

SQL
-- Lấy chi tiết hóa đơn của sản phẩm đắt nhất
SELECT *
FROM ChiTietHD
WHERE MaSP IN (SELECT MaSP FROM SanPham WHERE DonGia > 10000000);
-- Kết quả: chi tiết của sản phẩm 1 và 2

3.4 SUBQUERY VỚI EXISTS

Mục đích:

Kiểm tra subquery có trả về ít nhất một hàng hay không.

Cú pháp:

Cú pháp
WHERE EXISTS (subquery)

Ví dụ 1: EXISTS cơ bản

SQL
-- Lấy khách hàng có ít nhất một hóa đơn
SELECT *
FROM KhachHang kh
WHERE EXISTS (SELECT 1 FROM HoaDon hd WHERE hd.MaKH = kh.MaKH);

-- Tương đương với IN nhưng EXISTS thường nhanh hơn với dữ liệu lớn

Ví dụ 2: NOT EXISTS

SQL
-- Lấy khách hàng không có hóa đơn nào
SELECT *
FROM KhachHang kh
WHERE NOT EXISTS (SELECT 1 FROM HoaDon hd WHERE hd.MaKH = kh.MaKH);

-- Kết quả: khách hàng 5

Ví dụ 3: EXISTS với điều kiện phức tạp

SQL
-- Lấy khách hàng có mua sản phẩm giá > 20 triệu
SELECT DISTINCT kh.*
FROM KhachHang kh
WHERE EXISTS (
    SELECT 1
    FROM HoaDon hd
    JOIN ChiTietHD cthd ON hd.MaHD = cthd.MaHD
    JOIN SanPham sp ON cthd.MaSP = sp.MaSP
    WHERE hd.MaKH = kh.MaKH AND sp.DonGia > 20000000
);
-- Kết quả: khách hàng 1 và 2

3.5 CORRELATED SUBQUERY

Định nghĩa:

Correlated Subquery là subquery có tham chiếu đến các cột từ truy vấn bên ngoài.

Đặc điểm:

  • Chạy lại cho mỗi hàng của truy vấn bên ngoài
  • Thường chậm hơn non-correlated subquery

Ví dụ 1: Correlated Subquery với SELECT

SQL
-- Lấy thông tin mỗi hóa đơn và số hóa đơn của cùng khách hàng
SELECT
    hd1.MaHD,
    hd1.MaKH,
    hd1.TongTien,
    (SELECT COUNT(*)
     FROM HoaDon hd2
     WHERE hd2.MaKH = hd1.MaKH) AS SoHoaDonCuaKH
FROM HoaDon hd1;

Ví dụ 2: Correlated Subquery với WHERE

SQL
-- Lấy hóa đơn có tổng tiền lớn hơn tổng tiền trung bình của khách hàng đó
SELECT
    hd1.MaHD,
    hd1.MaKH,
    hd1.TongTien
FROM HoaDon hd1
WHERE hd1.TongTien > (
    SELECT AVG(hd2.TongTien)
    FROM HoaDon hd2
    WHERE hd2.MaKH = hd1.MaKH
);

3.6 SUBQUERY TRONG FROM

Mục đích:

Sử dụng kết quả của subquery như một bảng tạm (derived table).

Cú pháp:

Cú pháp
SELECT columns
FROM (subquery) AS alias;

Ví dụ 1: Subquery trong FROM

SQL
-- Tính tổng chi tiêu theo khách hàng, sau đó lấy trung bình
SELECT AVG(TongChiTieu) AS TB_ChiTieu
FROM (
    SELECT
        MaKH,
        SUM(TongTien) AS TongChiTieu
    FROM HoaDon
    GROUP BY MaKH
) AS ThongKe;

Ví dụ 2: Subquery trong FROM với JOIN

SQL
-- Lấy thông tin khách hàng và tổng chi tiêu
SELECT
    kh.TenKH,
    tk.TongChiTieu
FROM KhachHang kh
LEFT JOIN (
    SELECT
        MaKH,
        SUM(TongTien) AS TongChiTieu
    FROM HoaDon
    GROUP BY MaKH
) AS tk ON kh.MaKH = tk.MaKH;

3.7 SUBQUERY VỚI ANY VÀ ALL

Mục đích:

So sánh giá trị với bất kỳ (ANY) hoặc tất cả (ALL) giá trị từ subquery.

Cú pháp:

Cú pháp
WHERE column > ANY (subquery) -- Lớn hơn ít nhất một giá trị
WHERE column > ALL (subquery) -- Lớn hơn tất cả giá trị

Ví dụ 1: ANY

SQL
-- Lấy sản phẩm có giá lớn hơn ít nhất một sản phẩm của một loại nào đó
-- Ví dụ: sản phẩm có giá > ít nhất một sản phẩm giá rẻ
SELECT *
FROM SanPham
WHERE DonGia > ANY (SELECT DonGia FROM SanPham WHERE DonGia < 2000000);
-- Tương đương: > giá trị nhỏ nhất trong nhóm rẻ

Ví dụ 2: ALL

SQL
-- Lấy sản phẩm có giá lớn hơn tất cả sản phẩm giá rẻ
SELECT *
FROM SanPham
WHERE DonGia > ALL (SELECT DonGia FROM SanPham WHERE DonGia < 2000000);
-- Tương đương: > giá trị lớn nhất trong nhóm rẻ

PHẦN 4: VÍ DỤ TỔNG HỢP

4.1 Kết hợp Aggregate Functions, Joins và Subqueries

SQL - Phân tích doanh số bán hàng
-- 1. Doanh thu theo từng sản phẩm
SELECT
    sp.TenSP,
    SUM(cthd.SoLuong) AS SoLuongBan,
    SUM(cthd.SoLuong * cthd.DonGia) AS DoanhThu
FROM SanPham sp
LEFT JOIN ChiTietHD cthd ON sp.MaSP = cthd.MaSP
GROUP BY sp.MaSP, sp.TenSP
ORDER BY DoanhThu DESC;

-- 2. Top 3 sản phẩm bán chạy nhất
SELECT
    sp.TenSP,
    SUM(cthd.SoLuong) AS SoLuongBan
FROM SanPham sp
JOIN ChiTietHD cthd ON sp.MaSP = cthd.MaSP
GROUP BY sp.MaSP, sp.TenSP
ORDER BY SoLuongBan DESC
LIMIT 3;

-- 3. Khách hàng VIP (chi tiêu > trung bình)
SELECT
    kh.TenKH,
    SUM(hd.TongTien) AS TongChiTieu
FROM KhachHang kh
JOIN HoaDon hd ON kh.MaKH = hd.MaKH
GROUP BY kh.MaKH, kh.TenKH
HAVING SUM(hd.TongTien) > (SELECT AVG(TongTien) FROM HoaDon);

-- 4. Tháng có doanh thu cao nhất
SELECT
    MONTH(NgayLap) AS Thang,
    SUM(TongTien) AS DoanhThu
FROM HoaDon
GROUP BY MONTH(NgayLap)
ORDER BY DoanhThu DESC
LIMIT 1;

-- 5. Sản phẩm chưa từng bán được
SELECT *
FROM SanPham
WHERE MaSP NOT IN (SELECT DISTINCT MaSP FROM ChiTietHD);

-- 6. Khách hàng mua nhiều loại sản phẩm nhất
SELECT
    kh.TenKH,
    COUNT(DISTINCT cthd.MaSP) AS SoLoaiSP
FROM KhachHang kh
JOIN HoaDon hd ON kh.MaKH = hd.MaKH
JOIN ChiTietHD cthd ON hd.MaHD = cthd.MaHD
GROUP BY kh.MaKH, kh.TenKH
ORDER BY SoLoaiSP DESC
LIMIT 1;

PHẦN 5: TỔNG KẾT

5.1 BẢNG TÓM TẮT AGGREGATE FUNCTIONS

Hàm Mô tả Ví dụ
COUNT(*) Đếm số hàng SELECT COUNT(*) FROM HoaDon
SUM(column) Tính tổng SELECT SUM(TongTien) FROM HoaDon
AVG(column) Tính trung bình SELECT AVG(DonGia) FROM SanPham
MAX(column) Tìm lớn nhất SELECT MAX(TongTien) FROM HoaDon
MIN(column) Tìm nhỏ nhất SELECT MIN(TongTien) FROM HoaDon

5.2 BẢNG TÓM TẮT JOINS

Loại JOIN Mô tả Số hàng kết quả
INNER JOIN Chỉ lấy hàng khớp Ít hơn hoặc bằng
LEFT JOIN Tất cả bảng trái + khớp bảng phải Bằng số bảng trái
RIGHT JOIN Tất cả bảng phải + khớp bảng trái Bằng số bảng phải
CROSS JOIN Tích Descartes Bảng A * Bảng B

5.3 BẢNG TÓM TẮT SUBQUERIES

Loại Vị trí Đặc điểm
Scalar SELECT, WHERE, HAVING Trả về 1 giá trị
Table FROM Trả về bảng, cần alias
IN WHERE Kiểm tra trong tập hợp
EXISTS WHERE Kiểm tra tồn tại
Correlated WHERE, SELECT Tham chiếu bảng ngoài

Tóm tắt nhanh:

  • Aggregate Functions - Tổng hợp dữ liệu: SUM, AVG, COUNT, MAX, MIN
  • Joins - Kết hợp dữ liệu từ nhiều bảng: INNER, LEFT, RIGHT
  • Subqueries - Truy vấn lồng nhau để xử lý phức tạp
